void kmp_match(const char* T, const char* P)
{
    bool flag=false;
    int n=strlen(T);
    int m=strlen(P);
    int *next = new int[m];
    compute_prefix(P, next);
    int q=0;
    for (int i=0; i<n; i++)
    {
        while(q>0 && P[q]!=T[i])
            q=next[q-1];
        if(P[q]==T[i])
            q=q+1;
        if(q==m)
        {
            flag=true;
            cout<<"math occurs with "<<i-m+1<<endl;
            q=next[q-1];
        }
    }
    if(flag==false)
        cout<<"there is no math!"<<endl;
}
