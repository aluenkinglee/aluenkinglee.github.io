void deleteNode(ListNode * head, ListNode * node)
{
    /// 边界值
    if(node==NULL)
        return ;
    if(node->next!=NULL)
    {
        /// 把后面的节点值复制到本位置
        node->data = node->next->data;
        ListNode* temp = node->next;
        node->next = node->next->next;
        delete temp;
    } else {
    /// 删除的是最后一个节点，只能用遍历法
        ListNode* cur = head;
        while( cur->next != node)
            cur = cur->next;
        cur->next=NULL;
        delete node;
    }
}
