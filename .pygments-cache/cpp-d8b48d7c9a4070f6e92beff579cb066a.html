bool isIntersected(ListNode * head1, ListNode * head2)
{
    if(head1 == NULL || head2 == NULL)
        return false;
    ListNode* tail1=head1;
    ListNode* tail2=head2;
    while(tail1 != NULL)
        tail1 = tail1->next;
    while(tail2 != NULL)
        tail2 = tail2->next;
    ///若相交，尾节点可定相同
    return (tail1 == tail2);
}
