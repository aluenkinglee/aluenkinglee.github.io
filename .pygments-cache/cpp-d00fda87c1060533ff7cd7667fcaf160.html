/// 逆序打印
void RPrintList2(ListNode* head)
{
    stack<ListNode*> s;
    ListNode * tmp = head;
    while(tmp!=NULL)
    {
        s.push(tmp);
        tmp = tmp->next;
    }
    while(!s.empty())
    {
        tmp = s.top();
        cout << tmp->data<<"\t";
        s.pop();
    }
    cout << endl;
}
void RPrintList3(ListNode* head)
{
    if(head == NULL)
        return;
    RPrintList3(head->next);
    cout <<head->data<<"\t";
}
