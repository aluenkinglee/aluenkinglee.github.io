/// 将单链表反转
ListNode * reverseList(ListNode * head)
{
    /// 边界
    if(head==NULL || head->next == NULL)
        return head;
    ListNode* cur = head;
    /// 反转后的新链表头指针，初始为NULL
    ListNode* rHead = NULL;
    ListNode* temp  = NULL;
    while(cur != NULL)
    {
        temp = cur;
        cur = cur->next;
        /// 将废弃结点摘下，插入新链表的最前端
        temp->next = rHead;
        rHead = temp;
    }
    return rHead;
}
