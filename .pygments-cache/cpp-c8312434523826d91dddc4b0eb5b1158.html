///  查找单链表的中间结点
///  获取单链表中间结点，若链表长度为n(n>0)，则返回第n/2+1个结点
ListNode * getMidNode(ListNode * head )
{
    /// 边界值处理，空或者只有一个节点
    if(head == NULL || head->next == NULL)
        return head;
    ListNode* ahead = head;
    ListNode* behind = head;

    /// 注意判断条件
    while(ahead != NULL)    /// 前面指针每次走两步，直到指向最后一个结点，后面指针每次走一步
    {
        ahead = ahead->next;
        behind = behind->next;
        /// 注意判断条件
        if(ahead != NULL)
            ahead = ahead->next;
    }

    return behind;
}
